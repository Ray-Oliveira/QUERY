SELECT 
	SD1.D1_FILIAL,
	SD1.D1_COD,
	SD1.D1_UM,
	SD1.D1_QUANT,
	SD1.D1_VUNIT,
	SD1.D1_TOTAL,
	SD1.D1_CLASFIS,
	SB1.B1_ORIGEM
FROM
	SD1010 SD1
	INNER JOIN SB1010 SB1 ON SB1.B1_COD = SD1.D1_COD AND SB1.D_E_L_E_T_ = ''

WHERE
	SD1.D1_EMISSAO BETWEEN '20200401' AND '20200431'
	AND SD1.D_E_L_E_T_ = ''
	AND SUBSTRING(SD1.D1_CLASFIS,1,1) <> SB1.B1_ORIGEM

ORDER BY SD1.D1_COD
