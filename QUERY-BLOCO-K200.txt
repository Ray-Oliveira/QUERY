SELECT
	'K200',
	MAX(SB9.B9_DATA),
	cast(B9_COD as varchar(15)) + '01',
	CAST(SUM(B9_QINI) AS varchar)

FROM SB9010 SB9
	INNER JOIN SB1010 SB1 ON B1_COD = B9_COD AND SB1.D_E_L_E_T_ = ''
	
WHERE 
	B9_DATA = 20200430
	AND B9_QINI > 0
	AND SB1.B1_TIPO <> 'MO'
	AND SB1.B1_TIPO <> 'MC'
	AND SB1.B1_TIPO <> 'AT'
	AND SB1.B1_TIPO <> 'AI'
GROUP BY SB9.B9_COD
