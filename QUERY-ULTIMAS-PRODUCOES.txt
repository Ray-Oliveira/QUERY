SELECT TOP(3)
	SH6.H6_DTAPONT,
	SH6.H6_OP,
	SH6.H6_PRODUTO,
	SB1.B1_DESC,
	SH1.H1_DESCRI,
	SH6.H6_QTDPRO2,
	SH6.H6_DATAINI,
	SH6.H6_HORAINI,
	SH6.H6_DATAFIN,
	SH6.H6_HORAFIN

FROM SH6010 SH6
	INNER JOIN SB1010 SB1 ON SB1.B1_COD = SH6.H6_PRODUTO AND SB1.D_E_L_E_T_ = ''
	INNER JOIN SH1010 SH1 ON SH1.H1_CODIGO = SH6.H6_RECURSO AND SH1.D_E_L_E_T_ = ''

WHERE SH6.D_E_L_E_T_ = ''
AND SH6.H6_QTDPROD > 0
AND SB1.B1_X_GRUPO = 'TP0758'

ORDER BY SH6.H6_DTAPONT desc
